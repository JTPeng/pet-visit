# P2 用户体系 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现微信小程序登录、JWT 鉴权、手机号绑定、用户资料管理、关注关系，使后续社区和宠物档案模块有完整的用户基础。

**Architecture:** 后端新增 auth/user 两个模块，auth 负责微信登录和 JWT 签发/验证，user 负责资料和关注关系。全局 AuthGuard 保护所有接口，@Public 装饰器标记无需登录的接口。前端在 App 启动时静默登录，发帖/评论等操作前检查手机号绑定状态。

**Tech Stack:**
- 后端新增：@nestjs/jwt, @nestjs/passport, passport-jwt, axios（调微信 API）, @nestjs/throttler（限流）
- 前端新增：无新依赖（复用已有 Pinia + request 封装）

**追踪**：每完成一步，勾选本文件与 [README.md](./README.md) 中对应条目。

```typescript
import { Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { PassportStrategy } from '@nestjs/passport';
import { ExtractJwt, Strategy } from 'passport-jwt';

export interface JwtPayload {
  sub: string;
}

@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor(config: ConfigService) {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      ignoreExpiration: false,
      secretOrKey: config.get<string>('jwt.secret'),
    });
  }

  validate(payload: JwtPayload) {
    return { id: payload.sub };
  }
}
```

```typescript
import { ExecutionContext, Injectable } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { AuthGuard } from '@nestjs/passport';
import { IS_PUBLIC_KEY } from '../decorators/public.decorator';

@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {
  constructor(private reflector: Reflector) {
    super();
  }

  canActivate(context: ExecutionContext) {
    const isPublic = this.reflector.getAllAndOverride<boolean>(IS_PUBLIC_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (isPublic) return true;
    return super.canActivate(context);
  }
}
```

```typescript
export const CurrentUser = createParamDecorator(
  (data: keyof { id: string } | undefined, ctx: ExecutionContext) => {
```

```typescript
import { Injectable, UnauthorizedException, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { JwtService } from '@nestjs/jwt';
import axios from 'axios';
import { PrismaService } from '../prisma/prisma.service';

interface WxSessionResp {
  openid?: string;
  session_key?: string;
  unionid?: string;
  errcode?: number;
  errmsg?: string;
}

@Injectable()
export class AuthService {
  private readonly logger = new Logger(AuthService.name);

  constructor(
    private prisma: PrismaService,
    private jwt: JwtService,
    private config: ConfigService,
  ) {}

  async wxLogin(code: string) {
    const appid = this.config.get<string>('wx.appid');
    const secret = this.config.get<string>('wx.secret');
    const url = 'https://api.weixin.qq.com/sns/jscode2session';
    const { data } = await axios.get<WxSessionResp>(url, {
      params: { appid, secret, js_code: code, grant_type: 'authorization_code' },
      timeout: 5000,
    });

    if (!data.openid) {
      this.logger.warn(`wx code2session failed: ${JSON.stringify(data)}`);
      throw new UnauthorizedException(data.errmsg || '微信登录失败');
    }

    let user = await this.prisma.user.findUnique({ where: { openid: data.openid } });
    if (!user) {
      const nickname = `养宠人_${Math.floor(1000 + Math.random() * 9000)}`;
      user = await this.prisma.user.create({
        data: {
          openid: data.openid,
          unionid: data.unionid,
          nickname,
        },
      });
    }

    const token = await this.jwt.signAsync({ sub: user.id });
    return { token, user };
  }
}
```

- [ ] **Step 2: AuthService 增加 bindPhone 方法**

```typescript
private accessToken?: { value: string; expiresAt: number };

private async getAccessToken(): Promise<string> {
  if (this.accessToken && this.accessToken.expiresAt > Date.now()) {
    return this.accessToken.value;
  }
  const appid = this.config.get<string>('wx.appid');
  const secret = this.config.get<string>('wx.secret');
  const url = 'https://api.weixin.qq.com/cgi-bin/token';
  const { data } = await axios.get<{ access_token?: string; expires_in?: number; errmsg?: string }>(url, {
    params: { grant_type: 'client_credential', appid, secret },
    timeout: 5000,
  });
  if (!data.access_token) {
    throw new UnauthorizedException(data.errmsg || '获取 access_token 失败');
  }
  this.accessToken = {
    value: data.access_token,
    expiresAt: Date.now() + (data.expires_in ?? 7200) * 1000 - 60_000,
  };
  return data.access_token;
}

async bindPhone(userId: string, code: string) {
  const accessToken = await this.getAccessToken();
  const url = `https://api.weixin.qq.com/wxa/business/getuserphonenumber?access_token=${accessToken}`;
  const { data } = await axios.post<{ errcode: number; errmsg?: string; phone_info?: { purePhoneNumber: string } }>(
    url,
    { code },
    { timeout: 5000 },
  );
  if (data.errcode !== 0 || !data.phone_info) {
    throw new UnauthorizedException(data.errmsg || '获取手机号失败');
  }
  const phone = data.phone_info.purePhoneNumber;
  const user = await this.prisma.user.update({
    where: { id: userId },
    data: { phone },
  });
  return { user };
}
```

### P2-T8: Follow 模块

- [ ] **Step 1: 创建 FollowService**

新建 `serve/src/modules/follow/follow.service.ts`：

```typescript
import { BadRequestException, Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';

@Injectable()
export class FollowService {
  constructor(private prisma: PrismaService) {}

  async follow(followerId: string, followingId: string) {
    if (followerId === followingId) {
      throw new BadRequestException('不能关注自己');
    }
    const target = await this.prisma.user.findUnique({ where: { id: followingId } });
    if (!target) throw new NotFoundException('用户不存在');

    try {
      await this.prisma.follow.create({ data: { followerId, followingId } });
    } catch {
      // 已关注，幂等返回
    }
    return { ok: true };
  }

  async unfollow(followerId: string, followingId: string) {
    await this.prisma.follow.deleteMany({ where: { followerId, followingId } });
    return { ok: true };
  }

  async getFollowers(userId: string) {

      where: { followerId: userId },
      include: { following: { select: { id: true, nickname: true, avatar: true, bio: true } } },
      orderBy: { createdAt: 'desc' },
    });
    return list.map(item => item.following);
  }

  async isFollowing(followerId: string, followingId: string) {
    const found = await this.prisma.follow.findUnique({
      where: { followerId_followingId: { followerId, followingId } },
```

```typescript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import { wxLogin, bindPhone as bindPhoneApi } from '@/api/auth';
import { getMe, updateProfile as updateProfileApi, type UserProfile } from '@/api/user';

export const useUserStore = defineStore('user', () => {
  const token = ref<string>(getToken() || '');
  const profile = ref<UserProfile | null>(null);

  const isLoggedIn = computed(() => !!token.value);
  const hasPhone = computed(() => !!profile.value?.phone);

  async function login() {
    const { code } = await uni.login({ provider: 'weixin' });
    if (!code) throw new Error('微信 code 获取失败');
    const resp = await wxLogin(code);
    token.value = resp.token;
    setToken(resp.token);
    profile.value = resp.user;
    return resp.user;
  }

  async function fetchMe() {
    profile.value = await getMe();
    return profile.value;
  }

  async function updateProfile(data: Partial<Pick<UserProfile, 'nickname' | 'avatar' | 'bio'>>) {
    profile.value = await updateProfileApi(data);
    return profile.value;
  }

  async function bindPhone(code: string) {
    const { user } = await bindPhoneApi(code);
    profile.value = { ...(profile.value ?? {}), ...user };
    return profile.value;
  }

  function logout() {
    token.value = '';
    profile.value = null;
    removeToken();
  }

  return { token, profile, isLoggedIn, hasPhone, login, fetchMe, updateProfile, bindPhone, logout };
```
